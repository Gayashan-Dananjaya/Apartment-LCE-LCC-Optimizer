    CombRowVariable = 1
    
    Dim MonthlyVariable As Integer
    
    Set rngwall = Sheets("Alternative_Materials").Range("Wall_Alternatives[Wall Name]")
    Set rngwindow = Sheets("Alternative_Materials").Range("Window_Alternatives[Window Name]")
    Set rngdoor = Sheets("Alternative_Materials").Range("Door_Alternatives[Door Name]")
    Set rngfloor = Sheets("Alternative_Materials").Range("Floor_Alternatives[Floor Name]")
    Set rngroof = Sheets("Alternative_Materials").Range("Roof_Alternatives[Roof Name]")
    Set rngsystem = Sheets("Alternative_Materials").Range("System_Alternatives[AC System Name]")

    For Each cellwall In rngwall
        For Each cellwindow In rngwindow
            For Each celldoor In rngdoor
                For Each cellfloor In rngfloor
                    For Each cellroof In rngroof
                        For Each cellsystem In rngsystem
                            
                            Sheets("Wall").Range("D4") = cellwall.Offset(0, 2)
                            
                            'Wall U value assigning
                            Sheets("Cooling_Load_Calculation").Range("E15:E22") = cellwall.Offset(0, 1)
                            'Roof U value assgning
                            Sheets("Cooling_Load_Calculation").Range("E23") = cellroof.Offset(0, 1)
                            'Floor U value assigning
                            Sheets("Cooling_Load_Calculation").Range("D25") = cellfloor.Offset(0, 1)
                            'Door u value assigning
                            Sheets("Cooling_Load_Calculation").Range("D26") = celldoor.Offset(0, 1)
                            'Window U value assigning
                            Sheets("Cooling_Load_Calculation").Range("D31") = cellwindow.Offset(0, 2)

                            Sheets("Cooling_Load_Calculation").Range("F35:F42") = cellwindow.Offset(0, 1)

                            maxload = 0
                            
                            For MonthlyVariable = 1 To 12

                                Sheets("Window").Range("B1") = MonthlyVariable
                                Sheets("Wall").Range("D3") = MonthlyVariable
                                Sheets("Roof").Range("B3") = MonthlyVariable
                                
                                monthlytotalenergy = 0
                                dailytotalenergy = Sheets("Cooling_Load_Calculation").Range("C80")
                                
                                'Calculating monthly consuming energy
                                If MonthlyVariable = 1 Or MonthlyVariable = 3 Or MonthlyVariable = 5 Or MonthlyVariable = 7 Or MonthlyVariable = 8 Or MonthlyVariable = 10 Or MonthlyVariable = 12 Then
                                    monthlytotalenergy = dailytotalenergy * 31
                                ElseIf MonthlyVariable = 2 Then
                                    monthlytotalenergy = dailytotalenergy * 28
                                Else
                                    monthlytotalenergy = dailytotalenergy * 30
                                End If
                                
                                annualtotalenergy = annualtotalenergy + monthlytotalenergy
                                
                                'Assigning maximum cooling load
                                If Sheets("Cooling_Load_Calculation").Range("C81") > maxload Then
                                    maxload = Sheets("Cooling_Load_Calculation").Range("C81")
                                End If
                                                            
                            Next

                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 12) = annualtotalenergy * Sheets("Cooling_Load_Calculation").Range("K3") * 3.6
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 13) = (Sheets("Research_Wall").Range("R11") * cellwall.Offset(0, 6)) + (Sheets("Research_Window").Range("R11") * cellwindow.Offset(0, 5)) + (Sheets("Research_Doors").Range("G2") * celldoor.Offset(0, 4)) + (Sheets("Research_Floors").Range("E3") * cellfloor.Offset(0, 4)) + (Sheets("Research_Floors").Range("E2") * cellroof.Offset(0, 4))
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 14) = annualtotalenergy * Sheets("Cooling_Load_Calculation").Range("K3") * Sheets("Cooling_Load_Calculation").Range("K5")
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 15) = numberofacsystems * cellsystem.Offset(0, 2) * Application.WorksheetFunction.RoundUp((Sheets("Cooling_Load_Calculation").Range("K3")) / (Sheets("Cooling_Load_Calculation").Range("K4")), 0)
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 16) = ((Sheets("Research_Wall").Range("R11") * cellwall.Offset(0, 5)) + (Sheets("Research_Window").Range("R11") * cellwindow.Offset(0, 4)) + (Sheets("Research_Doors").Range("G2") * celldoor.Offset(0, 3)) + (Sheets("Research_Floors").Range("E3") * cellfloor.Offset(0, 3)) + (Sheets("Research_Floors").Range("E2") * cellroof.Offset(0, 3))) / 1000
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 17) = (annualtotalenergy * Sheets("Cooling_Load_Calculation").Range("K3") * Sheets("Cooling_Load_Calculation").Range("K2")) / 1000
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 18) = Sheets("Combination_Results").Cells(CombRowVariable + 1, 11) + Sheets("Combination_Results").Cells(CombRowVariable + 1, 12)
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 19) = (Sheets("Combination_Results").Cells(CombRowVariable + 1, 13) + Sheets("Combination_Results").Cells(CombRowVariable + 1, 14) + Sheets("Combination_Results").Cells(CombRowVariable + 1, 15)) / 1000000
                            Sheets("Combination_Results").Cells(CombRowVariable + 1, 20) = Sheets("Combination_Results").Cells(CombRowVariable + 1, 16) + Sheets("Combination_Results").Cells(CombRowVariable + 1, 17)

Sub Macro2()
'
' Macro2 Macro
'Orientation_Pro_Economic Cooling_Load_Calculation manual code typing----------------------------------------------------------------------------------------------------
    
    Dim cellwall As Range
    Dim cellwindow As Range
    Dim celldoor As Range
    Dim cellfloor As Range
    Dim cellroof As Range
    Dim cellsystem As Range

    Sheets.Add(After:=Worksheets("Orientation")).Name = "ORN_Pro_Economic"
    
    'Headers creation
    Sheets("ORN_Pro_Economic").Range("A1") = "No."
    Sheets("ORN_Pro_Economic").Range("B1") = "Wall Name"
    Sheets("ORN_Pro_Economic").Range("C1") = "Window Name"
    Sheets("ORN_Pro_Economic").Range("D1") = "Door Name"
    Sheets("ORN_Pro_Economic").Range("E1") = "Floor Name"
    Sheets("ORN_Pro_Economic").Range("F1") = "Roof Name"
    Sheets("ORN_Pro_Economic").Range("G1") = "AC System Name"
    Sheets("ORN_Pro_Economic").Range("H1") = "Maximum Load kWh/hr"
    Sheets("ORN_Pro_Economic").Range("I1") = "No. AC Systems"
    Sheets("ORN_Pro_Economic").Range("J1") = "Cooling Load (kWh/year)"
    Sheets("ORN_Pro_Economic").Range("K1") = "System Cost (LKR)"
    Sheets("ORN_Pro_Economic").Range("L1") = "Rotation Clockwise (Degrees)"
    Sheets("ORN_Pro_Economic").Range("M1") = "Operational Energy Reduction (Percentage)"
    Sheets("ORN_Pro_Economic").Range("N1") = "System Cost Reduction (Percentage)"
    
    Dim CombRowVariable As Integer
    Dim dailytotalenergy As Double
    Dim monthlytotalenergy As Double
    Dim annualtotalenergy As Double
    Dim maxload As Double
    Dim numberofacsystems As Double
    
    CombRowVariable = 1
    
    Dim MonthlyVariable As Integer
    Dim rotationvariable As Integer
    
    Set cellwall = Sheets("Orientation").Range("A5")
    Set cellwindow = Sheets("Orientation").Range("A7")
    Set celldoor = Sheets("Orientation").Range("A9")
    Set cellfloor = Sheets("Orientation").Range("A11")
    Set cellroof = Sheets("Orientation").Range("A13")
    Set cellsystem = Sheets("Orientation").Range("A15")
    
    For rotationvariable = 1 To 360

        Sheets("Wall").Range("D4") = cellwall.Offset(0, 2)
        
        'Wall U value assigning
        Sheets("Cooling_Load_Calculation").Range("E15:E22") = cellwall.Offset(0, 1)
        'Roof U value assgning
        Sheets("Cooling_Load_Calculation").Range("E23") = cellroof.Offset(0, 1)
        'Floor U value assigning
        Sheets("Cooling_Load_Calculation").Range("D25") = cellfloor.Offset(0, 1)
        'Door u value assigning
        Sheets("Cooling_Load_Calculation").Range("D26") = celldoor.Offset(0, 1)
        'Window U value assigning
        Sheets("Cooling_Load_Calculation").Range("D31") = cellwindow.Offset(0, 2)
        'Window SC assigning
        Sheets("Cooling_Load_Calculation").Range("F35:F42") = cellwindow.Offset(0, 1)
        
        annualtotalenergy = 0
        maxload = 0
        
        For MonthlyVariable = 1 To 12
            
            Sheets("City_Details").Range("E2") = MonthlyVariable
            Sheets("Window").Range("B1") = MonthlyVariable
            Sheets("Wall").Range("D3") = MonthlyVariable
            Sheets("Roof").Range("B3") = MonthlyVariable
            
            monthlytotalenergy = 0
            dailytotalenergy = Sheets("Cooling_Load_Calculation").Range("C80")
            
            'Calculating monthly consuming energy
            If MonthlyVariable = 1 Or MonthlyVariable = 3 Or MonthlyVariable = 5 Or MonthlyVariable = 7 Or MonthlyVariable = 8 Or MonthlyVariable = 10 Or MonthlyVariable = 12 Then
                monthlytotalenergy = dailytotalenergy * 31
            ElseIf MonthlyVariable = 2 Then
                monthlytotalenergy = dailytotalenergy * 28
            Else
                monthlytotalenergy = dailytotalenergy * 30
            End If
            
            annualtotalenergy = annualtotalenergy + monthlytotalenergy
            
            'Assigning maximum cooling load
            If Sheets("Cooling_Load_Calculation").Range("C81") > maxload Then
                maxload = Sheets("Cooling_Load_Calculation").Range("C81")
            End If
                                        
        Next
        
        numberofacsystems = Application.WorksheetFunction.RoundUp(maxload / cellsystem.Offset(0, 1), 0)
    
        'Value assigning
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 1) = CombRowVariable
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 2) = cellwall
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 3) = cellwindow
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 4) = celldoor
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 5) = cellfloor
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 6) = cellroof
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 7) = cellsystem
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 8) = maxload
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 9) = numberofacsystems
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 10) = annualtotalenergy
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 11) = numberofacsystems * cellsystem.Offset(0, 2) * Application.WorksheetFunction.RoundUp((Sheets("Cooling_Load_Calculation").Range("K3")) / (Sheets("Cooling_Load_Calculation").Range("K4")), 0)
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 12) = rotationvariable - 1
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 13) = (Sheets("Pro_Economic").Range("J2") - annualtotalenergy) * 100 / Sheets("Pro_Economic").Range("J2")
        Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 14) = (Sheets("Pro_Economic").Range("O2") - Sheets("ORN_Pro_Economic").Cells(CombRowVariable + 1, 11)) * 100 / Sheets("Pro_Economic").Range("O2")
        
        CombRowVariable = CombRowVariable + 1
        
        Sheets("Research_Wall").Range("J2") = Sheets("Research_Wall").Range("J2") + 1
        If Sheets("Research_Wall").Range("J2") >= 360 Then
            Sheets("Research_Wall").Range("J2") = Sheets("Research_Wall").Range("J2") - 360
        End If
        ActiveWorkbook.RefreshAll
    Next
